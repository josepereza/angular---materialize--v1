Initialization

// initialize a single carousel with an id
this.angular2MaterializeService.initCarousel('#carousel');

// initialize all carousels with class .carousel
this.angular2MaterializeService.initCarousel('.carousel');

// initialization with options
this.angular2MaterializeService.initCarousel('.carousel', {{'{'}}
  // insert options here
});

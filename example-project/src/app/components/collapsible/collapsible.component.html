Initialization

// initialize a single collapsible with an id
var instance = this.angular2MaterializeService.initCollapsible('#collapsible');

// initialize all collapsibles with class .collapsible
var instances = this.angular2MaterializeService.initCollapsible('.collapsible');

// initialization with options
var instances = this.angular2MaterializeService.initCollapsible('.collapsible', {{'{'}}
  // insert options here
});
